#include <iostream>
#include <cstring>
#include <iomanip>
using namespace std;

char *Mounth[] = {"Jr", "F", "Mr", "Ap", "M", "Jn", "J", "Au", "Sp", "Ok", "N", "D"};


int main()
{
int M = 12;
int Y = 3;
int** data = new int*[3];
for (int i = 0; i < Y; ++i)//создание двуиерного массива интов
{
    data[i] = new int[M];
}

for (int i = 0; i < M; ++i)//вывод массива чаров на экран
{
    cout << setw(3) << Mounth[i];

}
cout << endl;

for (int i = 0; i < Y; ++i)//вывод массива интов на экран
{
    cout << endl;
    for (int j = 0; j < M; ++j)
    {
        data[i][j] = 0;
        cout << setw(3) << data[i][j];
    }
    cout << endl;
}


for (int i = 0; i < Y; ++i)//вывод заполненного массва интов на экран
{
    cout << endl;
    for (int j = 0; j < M; ++j)
    {
        cout <<setw(3) << data[i][j];
    }
    cout << endl;
}

for (int i = 0; i < Y; ++i)//удаление интов
{
   delete [] data[i];
}

for (int i = 0; i < M; ++i)//удаление массива чаров
{
    Mounth[i] = NULL;
}



    return 0;
}

};

{

 cout << "Enter amount: ";
 cin >> Ct;
 cin.get();

 Car * ptr = new Car[Ct];

{
    cout << "Enter name: ";

for (int i = 0; i < Ct; ++i)
