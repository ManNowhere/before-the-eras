#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;


int main()
{
    int W; // ширина здания
    int H; // высота здания
    cin >> W >> H; cin.ignore();
    int N; // максимальное количество прыжков
    cin >> N; cin.ignore();
    int X0;//местоположение Бэтмена
    int Y0;
    cin >> X0 >> Y0; cin.ignore();
    int Xlast = W - 1;
    int Ylast = H - 1;
    int Xfirst = W - 1;
    int Yfirst = 0;
    bool flag = true;

    int e_num = 0;

    // game loop
    while (1) {
        string bombDir; // Current distance to the bomb compared to previous distance (COLDER, WARMER, SAME or UNKNOWN)
        cin >> bombDir; cin.ignore();
        if(bombDir == "COLDER")
        {
            e_num = 1;
        }
        else if(bombDir == "WARMER")
        {
            e_num = 2;
        }
        else if(bombDir == "SAME")
        {
            e_num = 3;
        }
        else if(bombDir == "UNKNOWN")
        {
            e_num = 4;
        }

        switch (e_num)
        {


        case 1:// COLDER стал дальше, чем до прыжка
        
            if((Ylast - Y0) <= 2)
                {
                    ++Yfirst ;
                }
                else
                {
                    Yfirst = Y0 + ((Ylast - Y0) / 2);
                }
                Y0 = Ylast;

            break;

        case 2://WARMER ближе, чем до прыжка
            if(flag == true)
            {
                if((Ylast - Y0) <= 2)
                {
                    ++Yfirst ;
                }
                Y0 = Yfirst; 
            }
            else
            {
                if((Xlast - X0) <= 2)
                    {
                        ++X0;
                    }
                    else
                    {
                        X0 += (Xlast - X0) / 2;
                    }
            }
        

            break;

        case 3://SAME находится на том же расстояни
        
            flag = false;
            Y0 = Yfirst + ((H - Yfirst) / 2);
            X0 = (W - Xfirst) / 2;


            break;

        case 4:// UNKNOWN расстояние неизвестно, встречается, только перед первым прыжком

            Yfirst = Y0;
            Ylast = H - 1;
            (Y0 <= (H / 2)) ? Y0 = H-1:Y0 = 0;
            // Yfirst = Y0;

            break;

        default:
            break;
        }


        cout << to_string(X0) << " " << to_string(Y0) << endl;

//        cout << "0 0" << endl;
    }







}
